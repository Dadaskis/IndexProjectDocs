Description

Graphics control module controls certain graphics settings of this game project.

Usage example

# Change antialiasing
# Possible values:
#  "disabled"
#  "x2"
#  "x4"
#  "x8"
# There was "x16" too, but it was removed because FUCK IT
Values.set_value("anti_aliasing_quality", "disabled")

# Whether enable or disable FXAA
Values.set_value("fxaa_enabled", false)

# Set sun shadow map resolution
# Possible values:
#  "very_low"
#  "low"
#  "medium"
#  "high"
#  "ultra"
Values.set_value("sun_shadow_resolution", "high")

# Shadow map filtering quality
# Possible values:
#  "low"
#  "medium"
#  "high"
Values.set_value("shadow_filter_quality", "high")

# Overall shadow map quality setting
# Possible values:
#  "low"
#  "medium"
#  "high"
#  "ultra"
Values.set_value("light_shadow_quality", "ultra")

# Depth map quality (whether HDR or not)
# Possible values:
#  "low"
#  "high"
Values.set_value("depth_quality", "high")

# Vertical Synchronization
# Possible values:
#  "disabled"
#  "enabled"
Values.set_value("vsync_enabled", "disabled")

# Depth prepass
# Possible values:
#  "disabled"
#  "enabled"
Values.set_value("depth_prepass", "enabled")

# Lighting quality, affects amount of lights one object can have
# Possible values:
#  "low"
#  "high"
Values.set_value("lighting_quality", "high")
